CREATE TABLE artistas (
  id INT PRIMARY KEY,
  nome VARCHAR(100)
);

CREATE TABLE albuns (
  id INT PRIMARY KEY,
  titulo VARCHAR(100),
  artista_id INT,
  FOREIGN KEY (artista_id) REFERENCES artistas(id)
);

CREATE TABLE faixas (
  id INT PRIMARY KEY,
  titulo VARCHAR(100),
  duracao INT,
  album_id INT,
  FOREIGN KEY (album_id) REFERENCES albuns(id)
);
